CREATE TABLE utenti (
    matricola VARCHAR(20) PRIMARY KEY,
    nome VARCHAR(50) NOT NULL,
    cognome VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    codice_fiscale CHAR(16) NOT NULL UNIQUE,
    password CHAR(128) NOT NULL,
    descrizione TEXT,
    foto_profilo LONGBLOB
);

CREATE TABLE contenuti (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titolo VARCHAR(255) NOT NULL,
    descrizione TEXT,
    tipo VARCHAR(50) NOT NULL,
    file_blob LONGBLOB NOT NULL,
    studente_email VARCHAR(255) NOT NULL,
    posizione int ,
    FOREIGN KEY (studente_email) REFERENCES utenti(email) ON DELETE CASCADE ON UPDATE CASCADE
);



CREATE TABLE otp_codes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    email VARCHAR(255) NOT NULL,
    otp_code VARCHAR(10) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_email_otp_created (email, otp_code, created_at DESC)
);

CREATE TABLE condivisioni (
    id INT AUTO_INCREMENT PRIMARY KEY,
    link_esterno VARCHAR(750) NOT NULL, -- Il link proveniente da Dropbox
    link_locale VARCHAR(750) NOT NULL UNIQUE,  -- Il link generato dalla tua app
    email_associata VARCHAR(191) NOT NULL,
    email_mittente VARCHAR(191) NOT NULL,
    data_condivisione TIMESTAMP DEFAULT null,
    visualized BOOLEAN DEFAULT false
);
